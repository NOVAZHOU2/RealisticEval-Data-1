```javascript
function numericalStrConvert(value) {
    /**
     * Convert the input string. First check if it's an integer, 
     * if so, convert it to an integer. If not, check if it is 
     * a floating point number. If yes, convert it to a float. 
     * If neither, return the original string.
     *
     * @param {string} value - Input value string
     * @returns {number|string} - Converted value
     */
    
    // Check if the value is an integer
    const intValue = parseInt(value, 10);
    if (!isNaN(intValue) && intValue.toString() === value) {
        return intValue;
    }
    
    // Check if the value is a float
    const floatValue = parseFloat(value);
    if (!isNaN(floatValue) && floatValue.toString() === value) {
        return floatValue;
    }
    
    // Return the original string if neither
    return value;
}
```

### Explanation:
- The function `numericalStrConvert` takes a string `value` as input.
- It first checks if `value` can be converted to an integer using `parseInt()`.
- If it's not an integer, it checks if it can be converted to a float using `parseFloat()`.
- If neither conversion is successful, it returns the original string.
