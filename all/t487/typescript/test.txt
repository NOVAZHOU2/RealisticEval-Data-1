Sure! Below is the equivalent TypeScript test case using Jest:

```typescript
import { describe, it, expect } from '@jest/globals';

function containsEmail(text: string): boolean {
    const emailRegex = /\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b/;
    return emailRegex.test(text);
}

describe('containsEmail function', () => {
    it('should return true if the text contains an email address', () => {
        expect(containsEmail('Please contact us at support@example.com')).toBe(true);
    });

    it('should return false if the text does not contain an email address', () => {
        expect(containsEmail('Hello, world!')).toBe(false);
    });
});
```

- We import `describe`, `it`, and `expect` from `@jest/globals`.
- We define the `containsEmail` function similarly to your Python function.
- We use `describe` to group related tests together.
- We use `it` to create individual test cases.
- We use `expect` to assert the expected outcomes of the function calls.
