```python
import re

/**
 * Verify that a string is a valid username and check that the username contains only letters, numbers, and underscores.
 *
 * @param {str} username - The username to be validated.
 * @return {bool} - Returns True if the username matches the regex pattern, indicating it is valid;
 *                  Returns False if the username contains any characters outside of letters, numbers, and underscores.
 */
def is_valid_username(username):
    """
    Verify that a string is a valid username and check that the username contains only letters, numbers, and underscores.

    :param str username: The username to be validated.
    :return: True if the username matches the regex pattern, indicating it is valid; False otherwise.
    :rtype: bool
    """
    # Regular expression to match a valid username (letters, numbers, and underscores)
    pattern = r'^[a-zA-Z0-9_]+$'
    
    # Check if the username matches the pattern
    return bool(re.match(pattern, username))
```
