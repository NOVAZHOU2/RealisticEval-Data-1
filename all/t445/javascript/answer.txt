Here is the equivalent JavaScript code using `mathjs` library for creating a rotation matrix:

```javascript
const math = require('mathjs');

function createRotMatrix(angleDeg, axis) {
    let angleRad = math.rad(angleDeg);
    let cosAngle = math.cos(angleRad);
    let sinAngle = math.sin(angleRad);

    if (axis === 'X') {
        return math.matrix([
            [1, 0, 0],
            [0, cosAngle, -sinAngle],
            [0, sinAngle, cosAngle]
        ]);
    } else if (axis === 'Y') {
        return math.matrix([
            [cosAngle, 0, sinAngle],
            [0, 1, 0],
            [-sinAngle, 0, cosAngle]
        ]);
    } else if (axis === 'Z') {
        return math.matrix([
            [cosAngle, -sinAngle, 0],
            [sinAngle, cosAngle, 0],
            [0, 0, 1]
        ]);
    } else {
        throw new Error("Invalid axis provided. Must be 'X', 'Y' or 'Z'.");
    }
}
```

In this JavaScript version, we use the `mathjs` library to handle matrix operations and trigonometric functions. The function takes an angle in degrees and an axis as input, converts the angle to radians, calculates the cosine and sine of the angle, and then returns a 4x4 rotation matrix based on the specified axis. If an invalid axis is provided, it throws an error.
