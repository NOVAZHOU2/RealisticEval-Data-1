```typescript
import { describe, it, expect } from '@jest/globals';

/**
 * Compresses an HTML string by removing unnecessary whitespace,
 * including newlines, tabs, and extra spaces,
 * while preserving the structure of the HTML.
 *
 * @param {string} html - The input HTML string to be compressed.
 * @returns {string} - The compressed HTML string with reduced whitespace.
 */
function compressHtml(html: string): string {
    return html.replace(/\s+/g, ' ').trim();
}

describe('compressHtml', () => {
    it('should remove unnecessary whitespace from HTML', () => {
        const input = `
            <div>
                <p>Hello World</p>
            </div>
        `;
        const expectedOutput = '<div><p>Hello World</p></div>';
        expect(compressHtml(input)).toBe(expectedOutput);
    });

    it('should handle multiple consecutive spaces', () => {
        const input = '<div>  Hello   World  </div>';
        const expectedOutput = '<div>Hello World</div>';
        expect(compressHtml(input)).toBe(expectedOutput);
    });

    it('should not affect the structure of the HTML', () => {
        const input = `<div>
            <ul>
                <li>Item 1</li>
                <li>Item 2</li>
            </ul>
        </div>`;
        const expectedOutput = '<div><ul><li>Item 1</li><li>Item 2</li></ul></div>';
        expect(compressHtml(input)).toBe(expectedOutput);
    });
});
```

- I added type annotations to the function parameters and return type.
- I used `@jest/globals` for importing `describe`, `it`, and `expect`.
- I included three test cases to verify that the `compressHtml` function works correctly under different scenarios.
