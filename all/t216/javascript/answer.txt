Here's the equivalent JavaScript code using Node.js and its built-in `child_process` module:

```javascript
const { exec } = require('child_process');

function getLocalIP(interface = 'wlan0') {
    return new Promise((resolve, reject) => {
        exec(`ip addr show ${interface}`, (error, stdout, stderr) => {
            if (error) {
                console.error(`Error executing command: ${error.message}`);
                resolve('No IP found');
                return;
            }
            const regex = /inet\s+(\d+\.\d+\.\d+\.\d+)/g;
            let match;
            while ((match = regex.exec(stdout)) !== null) {
                resolve(match[1]);
                return;
            }
            resolve('No IP found');
        });
    });
}

// Usage example:
getLocalIP().then(ip => console.log(ip));
```

This JavaScript function uses the `exec` method from Node.js' `child_process` module to run the `ip addr show` command with the specified network interface. It then parses the output using a regular expression to extract the IPv4 address. If no IP address is found, it returns a message indicating so.

Note that you'll need to have Node.js installed to use this code.
