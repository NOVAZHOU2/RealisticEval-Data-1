```javascript
// Import necessary library
const fs = require('fs');

/**
 * Reads numerical columns from a file starting from the line after the last line containing '/'.
 *
 * @param {string} fileName - The name of the file to read.
 * @returns {Array<Array<number>>} - A 2D array containing the numerical question.
 * @throws {Error} - If the file does not contain any '/' character.
 */
function readColumns(fileName) {
    // Read the file content
    let fileContent = fs.readFileSync(fileName, 'utf8');
    
    // Split the content into lines
    let lines = fileContent.split('\n');
    
    // Find the index of the last line containing '/'
    let lastIndex = -1;
    for(let i = 0; i < lines.length; i++) {
        if(lines[i].includes('/')) {
            lastIndex = i;
        }
    }

    // Throw an error if no '/' is found
    if(lastIndex === -1) {
        throw new Error("The file does not contain any '/' character.");
    }

    // Extract data from lines after the last line containing '/'
    let dataArray = [];
    for(let i = lastIndex + 1; i < lines.length; i++) {
        let row = lines[i].split(/\s+/).map(Number);
        dataArray.push(row);
    }

    return dataArray;
}
```

Please note that JavaScript does not have built-in support for NumPy arrays like Python, so I've used standard JavaScript Arrays instead. Also, Node.js's `fs` module is used for reading files synchronously. If you're running this in a browser environment, you would need to use asynchronous file reading methods or fetch API depending on your use case.
