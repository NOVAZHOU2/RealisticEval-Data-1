#include <string>
#include <utility> // For std::pair

// extract_sld_tld extracts the second-level and top-level domain names from the fully qualified domain name FQDN and returns them
// Args:
//     fqdn (std::string): The fully qualified domain name.
//
// Returns:
//     A pair containing the second-level domain and top-level domain.
std::pair<std::string, std::string> extract_sld_tld(const std::string& fqdn) {
    // Implementation of the function goes here
    // For example:
    size_t lastDot = fqdn.find_last_of('.');
    if (lastDot == std::string::npos) {
        return std::make_pair("", "");
    }

    size_t secondLastDot = fqdn.substr(0, lastDot).find_last_of('.');
    if (secondLastDot == std::string::npos) {
        return std::make_pair("", fqdn.substr(lastDot + 1));
    }

    std::string sld = fqdn.substr(secondLastDot + 1, lastDot - secondLastDot - 1);
    std::string tld = fqdn.substr(lastDot + 1);

    return std::make_pair(sld, tld);
}
