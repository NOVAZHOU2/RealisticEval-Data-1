#include <iostream>

#include <vector>

// Function to find and return a list of all placeholders in the format {{ placeholder }} from the input text.
std::vector<std::string> find_placeholders(const std::string& text) {
    // Use a regular expression to find placeholders in the specified format
    std::regex pattern(R"(\{\{\s*([\w]+)\s*\}\})");
    std::smatch match;
    std::vector<std::string> placeholders;

    auto begin = std::sregex_iterator(text.begin(), text.end(), pattern);
    auto end = std::sregex_iterator();

    for (std::sregex_iterator i = begin; i != end; ++i) {
        match = *i;
        placeholders.push_back(match[1].str());
    }

    return placeholders;
}

int main() {
    std::string sample_text = "This is a sample text with {{ placeholder1 }} and {{ placeholder2 }}.";
    std::vector<std::string> placeholders = find_placeholders(sample_text);

    std::cout << "Placeholders found:" << std::endl;
    for (const auto& placeholder : placeholders) {
        std::cout << placeholder << std::endl;
    }

    return 0;
}
