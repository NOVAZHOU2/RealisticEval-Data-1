Here's how you can convert your Python function into C++ code:

```c++
#include <string>
#include <algorithm>
#include <cctype>

bool isPhraseInStringIgnoreCase(const std::string &phrase, const std::string &input)
{
    // Convert both strings to lower case
    std::string lowerPhrase = phrase;
    std::transform(lowerPhrase.begin(), lowerPhrase.end(), lowerPhrase.begin(),
                   [](unsigned char c){ return std::tolower(c); });

    std::string lowerInput = input;
    std::transform(lowerInput.begin(), lowerInput.end(), lowerInput.begin(),
                  [](unsigned char c){ return std::tolower(c); });

    // Remove all white spaces from both strings
    lowerPhrase.erase(std::remove_if(lowerPhrase.begin(), lowerPhrase.end(), ::isspace), lowerPhrase.end());
    lowerInput.erase(std::remove_if(lowerInput.begin(), lowerInput.end(), ::isspace), lowerInput.end());

    // Check if the phrase exists in the input string
    size_t pos = lowerInput.find(lowerPhrase);
    return pos != std::string::npos;
}
```

This C++ code does the same thing as your Python function: it checks whether the phrase exists in the target string, ignoring case and allowing for variations in whitespace. Note that C++ doesn't have built-in support for regular expressions like Python, so we're using simple string manipulation functions instead.
