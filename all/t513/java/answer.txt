package org.real.temp;

import java.util.regex.Pattern;

public class Answer {

    /**
     * Check if the given phrase exists in the target string, ignoring case and allowing for variations in whitespace.
     *
     * @param phrase The phrase to search for in the string.
     * @param string The target string in which to search for the phrase.
     * @return True if the phrase is found as a whole word in the string, False otherwise.
     */
    public static boolean isPhraseInStringIgnoreCase(String phrase, String string) {
        // Convert both phrase and string to lower case
        String lowerCasePhrase = phrase.toLowerCase();
        String lowerCaseString = string.toLowerCase();

        // Replace all white spaces with regex pattern "\s+"
        Pattern pattern = Pattern.compile("\\s+");
        Matcher matcher = pattern.matcher(lowerCaseString);
        String cleanedString = matcher.replaceAll("");

        // Add word boundaries around the phrase
        String regexPattern = "\\b" + Pattern.quote(lowerCasePhrase) + "\\b";

        // Create a new Pattern object using the regular expression
        Pattern compiledPattern = Pattern.compile(regexPattern);

        // Create a new Matcher object using the compiled pattern
        Matcher phraseMatcher = compiledPattern.matcher(cleanedString);

        // Return true if the phrase is found in the string, false otherwise
        return phraseMatcher.find();
    }
}
