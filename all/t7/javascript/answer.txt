Sure! Below is the equivalent JavaScript code using the `winston` library, which is a popular logging library for Node.js that provides similar functionality to Python's `logging` module. If you don't have `winston` installed, you can install it using npm:

```bash
npm install winston
```

Here is the JavaScript code:

```javascript
const winston = require('winston');

class Logger {
    constructor(name, level = 'debug') {
        /**
         * Initializes a new logger instance.
         *
         * @param {string} name - Name of the logger, typically __filename to reference the module name.
         * @param {string} level - Logging level, default is 'debug'.
         */
        // Create logger
        this.logger = winston.createLogger({
            level: level,
            format: winston.format.combine(
                winston.format.timestamp(),
                winston.format.printf(info => {
                    return `${info.timestamp} - ${info.level.toUpperCase()} - ${info.filename}:${info.line}:${info.message}`;
                })
            ),
            transports: [
                new winston.transports.Console({
                    level: level,
                    format: winston.format.combine(
                        winston.format.colorize(),
                        winston.format.printf(info => {
                            return `${info.timestamp} - ${info.level.toUpperCase()} - ${info.filename}:${info.line}:${info.message}`;
                        })
                    )
                })
            ]
        });

        // Set filename and line number
        this.logger.filename = name;
        this.logger.line = 0; // This will be updated when logging
    }

    log(level, message) {
        /**
         * Logs a message with the given level.
         *
         * @param {string} level - Logging level for the message (e.g., 'debug', 'info').
         * @param {string} message - Log message.
         */
        switch (level) {
            case 'debug':
                this.logger.debug(message);
                break;
            case 'info':
                this.logger.info(message);
                break;
            case 'warn':
                this.logger.warn(message);
                break;
            case 'error':
                this.logger.error(message);
                break;
            case 'critical':
                this.logger.critical(message);
                break;
            default:
                this.logger.log(level, message);
                break;
        }
    }
}

// Example usage
const myLogger = new Logger(__filename, 'debug');
myLogger.log('debug', 'This is a debug message.');
myLogger.log('info', 'This is an info message.');
myLogger.log('warn', 'This is a warning message.');
myLogger.log('error', 'This is an error message.');
myLogger.log('critical', 'This is a critical message.');
```

### Explanation:
1. **Logger Class**: The `Logger` class is defined similarly to the Python version.
2. **Constructor**: The constructor initializes the logger with the specified name and level.
3. **Logging Levels**: The `log` method uses a switch statement to handle different logging levels.
4. **Winston Configuration**: The logger is configured with a custom format and a console transport.

Make sure to adjust the `filename` and `line` properties as needed for your specific use case.
