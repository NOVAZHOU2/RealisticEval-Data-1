```javascript
// Importing necessary modules
const { createLogger, format, transports } = require('winston');

/**
 * Represents a Logger class that handles logging messages.
 */
class Logger {
    /**
     * Initializes a new logger instance.
     *
     * @param {string} name - Name of the logger, typically __filename to reference the file name.
     * @param {number} [level=debug] - Logging level, default is debug.
     */
    constructor(name, level = 'debug') {
        // Creating a Winston logger instance
        this.logger = createLogger({
            level,
            format: format.combine(
                format.timestamp(),
                format.json()
            ),
            transports: [
                new transports.Console()
            ]
        });
    }

    /**
     * Logs a message with the given level.
     *
     * @param {string} level - Logging level for the message (e.g., 'info').
     * @param {string} message - Log message.
     */
    log(level, message) {
        // Logging the message using the appropriate level
        this.logger.log(level, message);
    }
}

module.exports = Logger;
```

This JavaScript code uses the `winston` library for logging, which provides a flexible and powerful logging solution. The `Logger` class initializes a logger with a specified name and level, and it has a method `log` to log messages at different levels.
