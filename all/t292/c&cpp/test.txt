```cpp
#define CATCH_CONFIG_MAIN  // This tells Catch to provide a main() - only do this in one cpp file
#include "catch.hpp"

// Assuming the function is implemented like this:
double calculateRemainingPayment(double principal, double interestRate, int numberOfPayments) {
    if (numberOfPayments == 0) return principal;
    double monthlyInterest = interestRate / 100.0;
    double payment = (principal * monthlyInterest * pow(1 + monthlyInterest, numberOfPayments)) /
                     (pow(1 + monthlyInterest, numberOfPayments) - 1);
    return payment;
}

TEST_CASE("Calculate Remaining Payment", "[loan]") {
    SECTION("Initial payment") {
        REQUIRE(calculateRemainingPayment(1000, 5, 12) == Approx(86.27));
    }

    SECTION("Multiple payments") {
        REQUIRE(calculateRemainingPayment(1000, 5, 24) == Approx(79.33));
    }

    SECTION("No payments") {
        REQUIRE(calculateRemainingPayment(1000, 5, 0) == 1000);
    }
}
```

In this example:
- We define the `calculateRemainingPayment` function with a simple implementation that calculates the monthly payment for a loan.
- We use Catch2's `TEST_CASE` macro to define our test cases.
- Each `SECTION` represents a different scenario or set of conditions for testing the function.
