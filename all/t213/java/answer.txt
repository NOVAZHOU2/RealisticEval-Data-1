package org.real.temp;

import java.util.Arrays;

public class Answer {

    public static int[][][] im2col(int[][][] image, int filterHeight, int filterWidth, int stride, int padding) {
        // Get dimensions of the input image
        int channels = image.length;
        int height = image[0].length;
        int width = image[0][0].length;

        // Calculate new dimensions after applying padding and stride
        int paddedHeight = height + 2 * padding;
        int paddedWidth = width + 2 * padding;
        int outputHeight = (paddedHeight - filterHeight) / stride + 1;
        int outputWidth = (paddedWidth - filterWidth) / stride + 1;

        // Create a new array for the output
        int[][][] col = new int[channels][filterHeight * filterWidth][outputHeight * outputWidth];

        // Initialize the padded image with zeros
        int[][][] paddedImage = new int[channels][paddedHeight][paddedWidth];
        for (int c = 0; c < channels; c++) {
            for (int i = 0; i < height; i++) {
                System.arraycopy(image[c][i], 0, paddedImage[c][padding + i], padding, width);
            }
        }

        // Perform the im2col operation
        int index = 0;
        for (int c = 0; c < channels; c++) {
            for (int h = 0; h <= paddedHeight - filterHeight; h += stride) {
                for (int w = 0; w <= paddedWidth - filterWidth; w += stride) {
                    for (int fH = 0; fH < filterHeight; fH++) {
                        for (int fW = 0; fW < filterWidth; fW++) {
                            col[c][index][h * stride + w] = paddedImage[c][h + fH][w + fW];
                        }
                    }
                    index++;
                }
            }
        }

        return col;
    }

    public static void main(String[] args) {
        // Example usage
        int[][][] image = {{{1, 2}, {3, 4}}, {{5, 6}, {7, 8}}};
        int[][][] result = im2col(image, 2, 2, 1, 1);
        for (int[][] channel : result) {
            for (int[] row : channel) {
                System.out.println(Arrays.toString(row));
            }
            System.out.println();
        }
    }
}
