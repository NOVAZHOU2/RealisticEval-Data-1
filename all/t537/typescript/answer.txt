```typescript
/**
 * Gets the current time formatted as 'hh:mm AM/PM'.
 *
 * @returns {string} The formatted time string.
 */
function getTime(): string {
    const now = new Date();
    let hours: number = now.getHours();
    let minutes: number = now.getMinutes();
    const ampm: string = hours >= 12 ? 'PM' : 'AM';

    // Convert hour to 12-hour format
    hours = hours % 12;
    hours = hours ? hours : 12; // If hours is 0, it should be 12

    // Add leading zero if necessary
    minutes = minutes < 10 ? '0' + minutes : minutes;

    return `${hours}:${minutes} ${ampm}`;
}
```

This TypeScript function includes type annotations for the return value and ensures that the variables `hours`, `minutes`, and `ampm` have appropriate types. It also handles the conversion of the time to a 12-hour format with leading zeros for minutes.
