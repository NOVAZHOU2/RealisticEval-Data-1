```typescript
/**
 * Move the emoji expressions in the string to the end of the text
 *
 * @param text - The input string containing text and possibly emojis.
 * @returns The modified string with all emojis moved to the end.
 */
function moveEmojisToEnd(text: string): string {
    // Regular expression to match emojis
    const emojiRegex = /[\u{1F600}-\u{1F64F}\u{1F300}-\u{1F5FF}\u{1F680}-\u{1F6FF}\u{2600}-\u{26FF}\u{2700}-\u{27BF}]/gu;

    // Extract emojis from the text
    const emojis = text.match(emojiRegex);

    // If no emojis found, return the original text
    if (!emojis) {
        return text;
    }

    // Remove emojis from the original text
    const cleanedText = text.replace(emojiRegex, '');

    // Concatenate the cleaned text with the emojis at the end
    return `${cleanedText}${emojis.join('')}`;
}
```

- **Regular Expression (`emojiRegex`)**: This regex matches most common emojis. It includes ranges for different categories of emojis such as smileys, people & body, animals & nature, food & drink, travel & places, activities, objects, symbols, flags, and skin tone modifiers.
  
- **Extracting Emojis**: We use `text.match(emojiRegex)` to extract all emojis from the input string.

- **Removing Emojis**: We remove the extracted emojis from the original text using `text.replace(emojiRegex, '')`.

- **Concatenating Text and Emojis**: Finally, we concatenate the cleaned text with the emojis joined together to form the final result.
