```typescript
class RandomStringGenerator {
    static LOWER_CASE: string = 'abcdefghijklmnopqrstuvwxyz';
    static UPPER_CASE: string = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
    static LENGTH: number = 10; // Set your desired length here

    static generateRandomString(): string {
        const randomString: string[] = [];

        // Ensure at least one upper case and one lower case letter
        randomString.push(this.getRandomCharacter(this.LOWER_CASE));
        randomString.push(this.getRandomCharacter(this.UPPER_CASE));

        // Fill the rest of the string length with random characters
        const allCharacters = this.LOWER_CASE + this.UPPER_CASE;
        for (let i = 2; i < this.LENGTH; i++) {
            randomString.push(this.getRandomCharacter(allCharacters));
        }

        // Shuffle the characters to ensure randomness
        return this.shuffleString(randomString.join(''));
    }

    private static getRandomCharacter(characters: string): string {
        const randomIndex = Math.floor(Math.random() * characters.length);
        return characters[randomIndex];
    }

    private static shuffleString(str: string): string {
        return str.split('').sort(() => Math.random() - 0.5).join('');
    }
}
```
