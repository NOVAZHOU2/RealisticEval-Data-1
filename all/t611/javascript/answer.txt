class StringGenerator {
    static LOWER_CASE = 'abcdefghijklmnopqrstuvwxyz';
    static UPPER_CASE = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
    static LENGTH = 10; // Set the desired length of the random string

    static generateRandomString() {
        // Use an array to construct the random string
        let randomString = [];

        // Ensure at least one upper case and one lower case letter
        randomString.push(this.getRandomCharacter(this.LOWER_CASE));
        randomString.push(this.getRandomCharacter(this.UPPER_CASE));

        // Fill the rest of the string length with random characters
        const allCharacters = this.LOWER_CASE + this.UPPER_CASE;
        for (let i = 2; i < this.LENGTH; i++) {
            const randomChar = this.getRandomCharacter(allCharacters);
            randomString.push(randomChar);
        }

        // Shuffle the characters to ensure randomness
        return this.shuffleString(randomString.join(''));
    }

    static getRandomCharacter(characters) {
        return characters[Math.floor(Math.random() * characters.length)];
    }

    static shuffleString(str) {
        return str.split('').sort(() => Math.random() - 0.5).join('');
    }
}
