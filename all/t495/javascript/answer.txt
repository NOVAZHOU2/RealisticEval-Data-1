```javascript
function filterContentWithContext(content, keywords, linesBefore = 1, linesAfter = 1) {
    // Convert keywords array to a single regex pattern
    const keywordPattern = new RegExp(`(^|\\s)(${keywords.join('|')})(?=\\s|$)`, 'gi');

    // Split content into lines
    let lines = content.split('\n');
    
    // Initialize an empty array to store the filtered content
    let resultLines = [];
    
    // Iterate over each line in the original content
    lines.forEach((line, index) => {
        // Check if the current line or any of its surrounding lines contain one of the keywords
        if (keywordPattern.test(line)) {
            // Add lines before the current line
            for (let i = Math.max(0, index - linesBefore); i < index; i++) {
                resultLines.push(lines[i]);
            }
            
            // Add the current line
            resultLines.push(line);
            
            // Add lines after the current line
            for (let i = index + 1; i <= Math.min(lines.length - 1, index + linesAfter); i++) {
                resultLines.push(lines[i]);
            }
        }
    });

    // Join the filtered lines back into a single string and return it
    return resultLines.join('\n');
}
```

This JavaScript function works similarly to your Python function. It searches through each line of the input `content` and checks if any of the provided `keywords` appear as whole words. If a keyword is found, it adds the specified number of lines before and after that line to the result.
