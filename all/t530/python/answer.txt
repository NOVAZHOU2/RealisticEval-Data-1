def create_matrix(rows, columns, initial_value):
    """
    Creates a matrix with the specified number of rows and columns,
    filled with the given initial value.

    :param int rows: The number of rows in the matrix.
    :param int columns: The number of columns in the matrix.
    :param initial_value: The value to fill the matrix with.
                         It can be of any type (int, str, dict, etc.).
    :return: A two-dimensional list (matrix) filled with the initial value.
    :rtype: list
    """
    return [[initial_value] * columns for _ in range(rows)]
