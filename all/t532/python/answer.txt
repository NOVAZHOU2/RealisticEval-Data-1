```python
def count_letter_changes(input_string):
    """
    Counts the occurrences of consecutive identical letters in a given string.

    :param input_string: The string to analyze for letter changes.
    :type input_string: str
    :return: An array of counts representing the number of consecutive
             identical letters before a different letter is found.
    :rtype: list[int]
    """
    if not input_string:
        return []

    counts = []
    current_count = 1

    for i in range(1, len(input_string)):
        if input_string[i] == input_string[i - 1]:
            current_count += 1
        else:
            counts.append(current_count)
            current_count = 1

    counts.append(current_count)

    return counts
```
