```cpp
#include <iostream>
#include <string>
#include <chrono>
#include <ctime>

/**
 * Calculate age based on the birth date string entered by the user.
 * Input birthDateString format such as "2000-01-01".
 * @param birthDateString
 * @return Age in years or -1 if the input is invalid.
 */
int calculateAge(const std::string& birthDateString) {
    // Parse the birth date string
    std::tm birthDate = {};
    strptime(birthDateString.c_str(), "%Y-%m-%d", &birthDate);

    // Get the current time
    auto now = std::chrono::system_clock::now();
    std::time_t nowTime = std::chrono::system_clock::to_time_t(now);
    std::tm* nowLocalTime = std::localtime(&nowTime);

    // Calculate the difference in years
    int age = nowLocalTime->tm_year + 1900 - (birthDate.tm_year + 1900);
    
    // Adjust the age if the birthday hasn't occurred yet this year
    if ((nowLocalTime->tm_mon < birthDate.tm_mon) ||
        (nowLocalTime->tm_mon == birthDate.tm_mon && nowLocalTime->tm_mday < birthDate.tm_mday)) {
        age--;
    }

    return age;
}

int main() {
    std::string birthDateString = "2000-01-01";
    int age = calculateAge(birthDateString);
    if (age != -1) {
        std::cout << "Age: " << age << std::endl;
    } else {
        std::cout << "Invalid birth date format." << std::endl;
    }
    return 0;
}
```

### Explanation:
1. **Parsing the Birth Date**: The `strptime` function is used to parse the birth date string into a `std::tm` structure.
2. **Getting Current Time**: The current time is obtained using `std::chrono::system_clock::now()` and converted to a `std::time_t` value, which is then converted to a local time using `std::localtime`.
3. **Calculating Age**: The age is calculated by subtracting the birth year from the current year. If the current month and day are before the birth month and day, the age is decremented by one.
4. **Main Function**: A simple `main` function demonstrates how to use the `calculateAge` function.
