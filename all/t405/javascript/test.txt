### Function Definition

```javascript
function removePartsOfString(...strings) {
    /**
     * Remove the part before the first upper case letter and the first lower case letter from the string.
     *
     * For example:
     *     input: 1234AbCde5678
     *     output: AbCde5678
     *
     * @param {...string} strings - Accepts one or more strings as variable arguments
     * @returns {Array<string>} - An array of modified strings
     */
    return strings.map(str => {
        const firstUpperCaseIndex = str.search(/[A-Z]/);
        const firstLowerCaseIndex = str.search(/[a-z]/);
        const startIndex = Math.min(firstUpperCaseIndex, firstLowerCaseIndex);
        return str.slice(startIndex === -1 ? 0 : startIndex);
    });
}

module.exports = removePartsOfString;
```
