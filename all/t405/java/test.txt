public class StringModifier {

    /**
     * Remove the part before the first uppercase letter and the first lowercase letter from the string.
     *
     * For example:
     *      input: "1234AbCde5678"
     *      output: "AbCde5678"
     *
     * @param strings Variable number of strings to process
     * @return Modified strings
     */
    public static String[] removePartsOfString(String... strings) {
        List<String> modifiedStrings = new ArrayList<>();
        for (String str : strings) {
            int firstUpperCaseIndex = -1;
            int firstLowerCaseIndex = -1;

            // Find the index of the first uppercase and lowercase characters
            for (int i = 0; i < str.length(); i++) {
                char ch = str.charAt(i);
                if (Character.isUpperCase(ch)) {
                    firstUpperCaseIndex = i;
                    break;
                }
            }

            for (int i = 0; i < str.length(); i++) {
                char ch = str.charAt(i);
                if (Character.isLowerCase(ch)) {
                    firstLowerCaseIndex = i;
                    break;
                }
            }

            // Determine the starting index for substring
            int startIndex = Math.min(firstUpperCaseIndex, firstLowerCaseIndex);
            if (startIndex != -1) {
                modifiedStrings.add(str.substring(startIndex));
            } else {
                modifiedStrings.add(""); // If no uppercase or lowercase character is found
            }
        }

        return modifiedStrings.toArray(new String[0]);
    }
}
