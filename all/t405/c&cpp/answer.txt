// Function to find the index of the first character that satisfies the predicate
template<typename Predicate>
int find_first_index(const std::string& str, Predicate pred) {
    for (size_t i = 0; i < str.size(); ++i) {
        if (pred(str[i])) {
            return static_cast<int>(i);
        }
    }
    return -1; // Return -1 if no matching character is found
}

// Function to process and modify strings based on the specified criteria
std::vector<std::string> remove_parts_of_string(const std::vector<std::string>& strings) {
    std::vector<std::string> results;
    
    for (const auto& string : strings) {
        std::string modifiedString = string;

        // Remove all characters before the first uppercase letter
        int indexUpper = find_first_index(modifiedString, ::isupper);
        if (indexUpper != -1) {
            modifiedString = modifiedString.substr(indexUpper);
        }

        int indexLower = find_first_index(modifiedString, ::islower);
        if (indexLower != -1) {
            modifiedString = modifiedString.substr(indexLower - 1);
        }

        results.push_back(modifiedString);
    }

    return results;
}

int main() {

}
