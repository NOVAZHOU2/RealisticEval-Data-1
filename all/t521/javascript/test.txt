const { wordFilterCounter } = require('./path_to_your_module');

describe('wordFilterCounter', () => {
  it('counts the occurrences of specified words in the given text', () => {
    const text = 'hello world hello';
    const filterWords = ['hello', 'world'];
    expect(wordFilterCounter(text, filterWords)).toEqual({'hello': 2, 'world': 1});
  });

  it('returns an empty object if no filter words are provided', () => {
    const text = 'hello world hello';
    const filterWords = [];
    expect(wordFilterCounter(text, filterWords)).toEqual({});
  });

  it('ignores non-matching words', () => {
    const text = 'hello world hello';
    const filterWords = ['goodbye', 'universe'];
    expect(wordFilterCounter(text, filterWords)).toEqual({});
  });
});
