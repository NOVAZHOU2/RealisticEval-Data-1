    let dx = Math.abs(x2 - x1);
    let dy = -Math.abs(y2 - y1);

        points.push([x1, y1]);
        if (x1 === x2 && y1 === y2) {

        }

        // horizontal step
        if (e2 >= dy) {
            if (x1 === x2) {
                break;
            }
            err += dy;

        }

        // vertical step

                break;
            }
            err += dx;
            y1 += sy;
        }
    }

    return points;
}
