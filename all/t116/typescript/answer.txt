    let dx = thisPoint.x - otherPoint.x;
    let dy = thisPoint.y - otherPoint.y;

    // Handle wraparound for the x dimension
    if (Math.abs(dx) > width / 2) {
        dx = dx > 0 ? dx - width : dx + width;
