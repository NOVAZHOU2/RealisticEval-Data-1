Here is an equivalent C++ code for your Python function:

```c++
#include <fstream>
#include <string>
#include <iostream>
#include <nlohmann/json.hpp>

using namespace std;
using json = nlohmann::json;

vector<json> read_txt_add_json_bracket(const string &filename) {
    ifstream file(filename);
    if (!file.is_open()) throw runtime_error("File not found");

    string content((istreambuf_iterator<char>(file)), istreambuf_iterator<char>());
    file.close();

    try {
        // Wrap content with JSON brackets
        string jsonContent = "[" + content + "]";

        // Parse JSON
        vector<json> result = json::parse(jsonContent);

        return result;
    } catch (json::exception &e) {
        cerr << "JSON parsing error: " << e.what() << endl;
        throw runtime_error("Invalid JSON format");
    }
}
```

Note that this C++ code uses the nlohmann/json library to parse JSON. If you don't have this library installed, you can get it via `vcpkg install nlohmann-json` or download it from [here](https://github.com/nlohmann/json).
