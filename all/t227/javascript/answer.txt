const fs = require('fs');
const path = require('path');

function countUniqueColors(imagePath) {
    /**
     * Count the number of unique colors in an image.
     *
     * @param {string} imagePath - Path to the image file.
     * @returns {Promise<number>} - The number of unique colors in the image.
     */
    
    return new Promise((resolve, reject) => {
        const imageBuffer = fs.readFileSync(imagePath);
        
        // Using sharp library to handle image processing
        const sharp = require('sharp');
        sharp(imageBuffer)
            .raw()
            .toBuffer()
            .then(data => {
                const width = 100; // Assuming a fixed width for simplicity
                const height = data.length / (width * 3); // Assuming RGB format
                const colorSet = new Set();
                
                for (let i = 0; i < data.length; i += 3) {
                    const r = data[i];
                    const g = data[i + 1];
                    const b = data[i + 2];
                    const colorKey = `${r},${g},${b}`;
                    colorSet.add(colorKey);
                }
                
                resolve(colorSet.size);
            })
            .catch(err => {
                reject(err);
            });
    });
}

// Example usage:

    });

### Explanation:
1. **Reading the Image**: The image is read synchronously using `fs.readFileSync`.
2. **Image Processing with Sharp**: The `sharp` library is used to process the image and extract raw pixel data.
3. **Extracting Colors**: The pixel data is iterated over in chunks of three (representing RGB values), and each unique color is added to a `Set` to ensure uniqueness.
4. **Returning the Result**: The size of the `Set` (number of unique colors) is returned.
