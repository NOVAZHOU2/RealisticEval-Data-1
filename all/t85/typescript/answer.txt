Here is the equivalent TypeScript code using the 'pandas' library:

```typescript
import * as pandas from 'pandas';

function fillMissingWithFirstValid(df: pandas.DataFrame, columnName: string): pandas.DataFrame {
    /**
     * Fills missing values in the specified column of the DataFrame with the first valid value in that column.
     * @param {pandas.DataFrame} df - The pandas DataFrame.
     * @param {string} columnName - The name of the column to fill missing values.
     * @returns {pandas.DataFrame} - The DataFrame with missing values filled.
     */
    let firstValidValue = df[columnName].dropna().iloc[0];
    return df.fillna({ [columnName]: firstValidValue });
}
```
