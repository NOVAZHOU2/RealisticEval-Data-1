Here's the equivalent Java code for your Python function using the Apache Commons CSV library:

```java
package org.real.temp;

import java.io.IOException;
import java.util.Iterator;
import org.apache.commons.csv.CSVFormat;
import org.apache.commons.csv.CSVParser;
import org.apache.commons.csv.CSVRecord;

public class Answer {

    public static void fillMissingWithFirstValid(String filePath, String columnName) throws IOException {
        try (CSVParser parser = new CSVParser(java.nio.file.Files.newBufferedReader(new java.io.File(filePath)), 
                CSVFormat.DEFAULT.withHeader())) {
            
            Iterator<CSVRecord> records = parser.iterator();
            if (!records.hasNext()) return; // Check if file is empty
            
            String firstValidValue = null;
            while (records.hasNext() && firstValidValue == null) {
                CSVRecord record = records.next();
                firstValidValue = record.get(columnName);
                if (!firstValidValue.isEmpty()) break;
            }
            
            if (firstValidValue != null) {
                records.forEachRemaining(record -> {
                    if (record.get(columnName).isEmpty()) {
                        record.set(columnName, firstValidValue);
                    }
                });
            }
        }
    }
}
```

Please note that this code assumes you're working with CSV files and not pandas DataFrames. If you want to work with pandas DataFrames in Java, you might consider using libraries like Jupyter or Spark.
