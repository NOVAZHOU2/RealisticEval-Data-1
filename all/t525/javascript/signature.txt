/**
 * Flip the point cloud across a specified axis.
 *
 * @param {Array<Array<number>>} pointCloud - A N x 3 array representing the 3D point cloud.
 * @param {number} axis - An integer specifying the axis to flip (0 for x, 1 for y, 2 for z).
 * @returns {Array<Array<number>>} - A N x 3 array of the flipped point cloud.
 */
function flipPointCloud(pointCloud, axis) {
    // Create a new array to store the flipped point cloud
    const flippedPointCloud = [];

    // Iterate over each point in the original point cloud
    for (let i = 0; i < pointCloud.length; i++) {
        const point = pointCloud[i];

        // Clone the current point to avoid modifying the original array
        let flippedPoint = [...point];

        // Flip the point across the specified axis
        if (axis === 0) {
            flippedPoint[0] = -flippedPoint[0];
        } else if (axis === 1) {
            flippedPoint[1] = -flippedPoint[1];
        } else if (axis === 2) {
            flippedPoint[2] = -flippedPoint[2];
        }

        // Add the flipped point to the new array
        flippedPointCloud.push(flippedPoint);
    }

    return flippedPointCloud;
}
