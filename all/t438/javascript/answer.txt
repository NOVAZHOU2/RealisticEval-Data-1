Here's how you can convert your Python code to JavaScript using Node.js with 'pandas-js' library:

```javascript
const fs = require('fs');
const csvtojson = require('csvtojson');

async function readCSVToDataFrame(filePath) {
    try {
        let jsonArray = await csvtojson().fromFile(filePath);
        return jsonArray;
    } catch(err) {
        console.error(`Error reading CSV file: ${err}`);
        throw err;
    }
}

// Usage
readCSVToDataFrame('/path/to/your/file.csv')
    .then(data => {
        // Now data is an array of JSON objects
        console.log(data);
    })
    .catch(error => {
        console.error(error);
    });
```

Note that 'pandas-js' is not an official package from pandas but a community-driven one. It may not support all features of pandas, especially those related to complex data manipulation or analysis.

If you need full pandas functionality in JavaScript, you might consider using Python on the backend server and calling its API from the frontend JavaScript code via AJAX or Fetch API.
