/**
 * Count the number of letters in a string.
 *
 * @param str - The input string from which to count letters.
 * @return int - The count of letters in the string.
 */
public class StringCounter {
    public static int countLetters(String str) {
        if (str == null) {
            return 0;
        }
        
        int letterCount = 0;
        for (int i = 0; i < str.length(); i++) {
            char c = str.charAt(i);
            if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')) {
                letterCount++;
            }
        }
        
        return letterCount;
    }

    public static void main(String[] args) {
        // Example usage:
        String testString = "Hello, World!";
        int result = countLetters(testString);
        System.out.println("Number of letters: " + result); // Output: Number of letters: 10
    }
}
