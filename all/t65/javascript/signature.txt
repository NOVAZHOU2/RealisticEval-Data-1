/**
 * Find duplicate IPs in the given IP list excluding specified IPs to ignore.
 *
 * @param {string[]} ipList - List of IP addresses
 * @param {string[]} ignoreList - List of IP addresses to ignore
 * @returns {string[]} A list of duplicate IPs excluding those in the ignore list.
 */
function findDuplicateIps(ipList, ignoreList) {
    // Create a set to store unique non-ignored IPs
    const uniqueNonIgnoredIps = new Set();
    
    // Create a set to store duplicates
    const duplicates = new Set();

    // Iterate through each IP in the list
    for (const ip of ipList) {
        if (!ignoreList.includes(ip)) {
            if (uniqueNonIgnoredIps.has(ip)) {
                // If the IP is already in the set, it's a duplicate
                duplicates.add(ip);
            } else {
                // Otherwise, add it to the set
                uniqueNonIgnoredIps.add(ip);
            }
        }
    }

    // Convert the duplicates set to an array before returning
    return Array.from(duplicates);
}
