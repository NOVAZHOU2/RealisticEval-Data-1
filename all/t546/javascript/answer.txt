Here's the equivalent JavaScript code using Node.js:

```javascript
const readline = require('readline');

function readTsvFromStdin() {
  const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
  });

  let lines = [];
  let maxLength = 0;

  rl.on('line', (line) => {
    let fields = line.split('\t');
    if (fields.length > maxLength) {
      maxLength = fields.length;
    }
    while (fields.length < maxLength) {
      fields.push('');
    }
    lines.push(fields);
  });

  rl.on('close', () => {
    console.log(lines);
  });
}

readTsvFromStdin();
```

This JavaScript code reads tab-separated values (TSV) from standard input, similar to your Python code. It uses the `readline` module to handle the input and processes it line by line. The `lines` array stores the processed TSV data, and `maxLength` keeps track of the maximum number of fields in any given line to ensure proper padding. When the input ends (`rl.on('close')`), the function logs the processed data to the console.
