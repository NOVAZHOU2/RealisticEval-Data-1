function translatePointCloud(pointCloud: number[][], translationVector: number[]): number[][] {
    return pointCloud.map((point) => [
        point[0] + translationVector[0],
        point[1] + translationVector[1],
        point[2] + translationVector[2]
    ]);
}

This function accepts two arguments: `pointCloud` and `translationVector`. The former should be an array of arrays, each representing a 3D point in space. The latter should be an array representing the translation vector. It returns a new array of points after applying the translation vector to each point in the original array.
