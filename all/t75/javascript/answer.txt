const fs = require('fs');
const path = require('path');

function renameFiles(directory) {
    /**
     * Renames PNG files in a specified directory by appending a sequence number to each file.
     * The files are sorted alphabetically, and each base name is assigned sequential numbers.
     * For example:
     *     If the directory contains three PNG files named "image1.png", "image2.png", "image3.png",
     *     after renaming, the files will be named "image1001.png", "image2001.png", "image3001.png".
     *
     * @param {string} directory - The path to the directory containing PNG files to be renamed.
     */

    // Read all files in the directory
    fs.readdir(directory, (err, files) => {
        if (err) {
            return console.error('Error reading directory:', err);
        }

        // Filter out only PNG files
        const pngFiles = files.filter(file => path.extname(file).toLowerCase() === '.png');

        // Sort the PNG files alphabetically
        pngFiles.sort();

        let counter = 1001; // Start counting from 1001

        // Rename each PNG file
        pngFiles.forEach((file, index) => {
            const oldPath = path.join(directory, file);
            const newFileName = `image${counter}.png`;
            const newPath = path.join(directory, newFileName);

            fs.rename(oldPath, newPath, err => {
                if (err) {
                    return console.error(`Error renaming ${oldPath} to ${newPath}:`, err);
                }
                console.log(`${oldPath} renamed to ${newPath}`);
                counter++;
            });
        });
    });
}
