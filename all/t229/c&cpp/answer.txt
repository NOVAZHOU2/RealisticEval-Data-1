Here is the equivalent C++ code for your Python function:

```c++
#include <iostream>
#include <string>

std::string convertFileSize(int sizeBytes)
{
    if (sizeBytes == 0)
        return "0B";
    
    std::string units[] = {"B", "KB", "MB", "GB", "TB"};
    int unitIndex = 0;

    while (sizeBytes >= 1024 && unitIndex + 1 < sizeof(units)/sizeof(std::string))
    {
        sizeBytes /= 1024;
        unitIndex++;
    }

    return std::to_string(sizeBytes) + units[unitIndex];
}

int main()
{
    std::cout << convertFileSize(2120) << std::endl; // Output: 2KB
    return 0;
}
```

This C++ version does exactly what your Python function does. It takes an integer representing the size of a file in bytes and returns a string with the size in a more readable form (e.g., "2KB", "1MB").
