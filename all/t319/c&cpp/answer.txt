#include <iostream>
#include <string>

/**
 * Count the number of dashes in a string
 *
 * @param str - The string from which to count dash characters.
 * @return The total count of dash characters found in the string.
 */
int countDashes(const std::string& str) {
    int count = 0;
    for (char ch : str) {
        if (ch == '-') {
            count++;
        }
    }
    return count;
}

int main() {
    // Example usage
    std::string input = "Hello-World-this-is-a-test-string";
    int result = countDashes(input);
    std::cout << "Number of dashes: " << result << std::endl;
    return 0;
}
