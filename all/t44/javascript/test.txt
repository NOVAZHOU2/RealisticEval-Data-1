### JavaScript Function Definition (`alignLinesLeft.js`)
```javascript
function alignLinesLeft(str1, str2) {
    // Calculate the maximum length of the two strings
    const maxLength = Math.max(str1.length, str2.length);
    
    // Pad both strings to the right with spaces to match the maximum length
    const paddedStr1 = str1.padEnd(maxLength);
    const paddedStr2 = str2.padEnd(maxLength);
    
    return [paddedStr1, paddedStr2];
}

module.exports = alignLinesLeft;
```
