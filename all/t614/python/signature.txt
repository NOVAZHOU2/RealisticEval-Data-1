from typing import List

def calculate_average_difference(numbers: List[int]) -> float:
    if len(numbers) < 2:
        return 0.0

    total_difference = 0
    for i in range(1, len(numbers)):
        total_difference += abs(numbers[i] - numbers[i - 1])
    
    average_difference = total_difference / (len(numbers) - 1)
    return average_difference
