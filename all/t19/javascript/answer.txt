```javascript
function containsPhoneNumber(s) {
    /**
     * Determines whether the string contains a phone number.
     * A possible format for a phone number is +1-800-555-1234, 555-555-1234, or 555 555 1234.
     * 
     * @param {string} s - The input string which may contain a phone number.
     * @returns {boolean} - Whether this string contains a phone number.
     */

    // Regular expression to match phone numbers in the specified formats
    const phoneNumberRegex = /(\+1-\d{3}-\d{3}-\d{4})|(\d{3}-\d{3}-\d{4})|(\d{3} \d{3} \d{4})/;

    return phoneNumberRegex.test(s);
}

// Example usage:
console.log(containsPhoneNumber("+1-800-555-1234")); // true
console.log(containsPhoneNumber("555-555-1234"));   // true
console.log(containsPhoneNumber("555 555 1234"));   // true
console.log(containsPhoneNumber("Hello World!")); // false
```

This JavaScript function uses a regular expression (`phoneNumberRegex`) to check if the input string `s` contains a phone number in any of the specified formats. The `test` method of the regular expression object is used to determine if there is a match. If a match is found, the function returns `true`; otherwise, it returns `false`.
