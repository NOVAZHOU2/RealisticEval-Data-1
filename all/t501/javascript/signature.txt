/**
 * Converts a string concatenated with underscores to a short format.
 * For example:
 *     input: "f1_p1_g1_b1_c1"
 *     output: "fpgbc"
 * 
 * @param {string} inputStr - The input string with segments separated by underscores.
 * 
 * @returns {string} - A short format string derived from the first characters of each segment.
 */
function convertToShortFormat(inputStr) {
    // Split the input string by underscores into an array of segments
    const segments = inputStr.split('_');
    
    // Initialize an empty array to store the first character of each segment
    let result = [];
    
    // Iterate over each segment in the array
    for (let i = 0; i < segments.length; i++) {
        // Add the first character of the current segment to the result array
        result.push(segments[i][0]);
    }
    
    // Join the result array into a single string and return it
    return result.join('');
}
