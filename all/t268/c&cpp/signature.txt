#include <vector>
#include <limits>

// Function to determine if there exists a starting gas station's index 
// where you can travel around the circuit once in a clockwise direction.
// 
// Args:
//     gas: A vector of integers representing the amount of gas at each station.
//     cost: A vector of integers representing the cost of gas to travel from each station to the next.
// 
// Returns:
//     The starting gas station's index if the circuit can be completed, otherwise -1.
int canCompleteCircuit(const std::vector<int>& gas, const std::vector<int>& cost) {
    int totalGas = 0;
    int totalCost = 0;
    int start = 0;
    int currentGas = 0;

    for (size_t i = 0; i < gas.size(); ++i) {
        totalGas += gas[i];
        totalCost += cost[i];
        currentGas += gas[i] - cost[i];

        if (currentGas < 0) {
            start = i + 1;
            currentGas = 0;
        }
    }

    if (totalGas < totalCost) {
        return -1;
    } else {
        return start;
    }
}
