#include <iostream>
#include <string>
#include <sstream>

std::string snakeToCamel(const std::string& snakeStr) {
    std::stringstream ss(snakeStr);
    std::string segment;
    std::string camelStr;

    while (std::getline(ss, segment, '_')) {
        if (!segment.empty()) {
            // Capitalize the first letter of each segment and append it to the result
            segment[0] = toupper(segment[0]);
            camelStr += segment;
        }
    }

    return camelStr;
}

int main() {
    std::string snakeString = "example_string";
    std::cout << snakeToCamel(snakeString) << std::endl;  // Output: ExampleString
    return 0;
}
