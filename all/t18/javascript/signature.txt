/**
 * Convert a floating point number between 0 and 1 to a color from red to green in the RGB color format.
 * @param {number} value - A float between 0 and 1.
 * @returns {Array<number>} A tuple representing the RGB color.
 */
function floatToRgb(value) {
    // Ensure the input value is within the valid range [0, 1]
    if (value < 0 || value > 1) {
        throw new Error('Value must be between 0 and 1');
    }

    // Calculate the red and green components based on the value
    let red = Math.floor(255 * (1 - value));
    let green = Math.floor(255 * value);

    // The blue component remains constant at 0 for the given range
    let blue = 0;

    return [red, green, blue];
}

This JavaScript function `floatToRgb` takes a floating-point number between 0 and 1 and returns an array representing the RGB color. The red component decreases linearly from 255 to 0 as the value increases from 0 to 1, while the green component increases linearly from 0 to 255. The blue component remains at 0 throughout the range.
