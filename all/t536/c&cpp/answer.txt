#include <iostream>
#include <iomanip>
#include <ctime>

/**
 * Gets the current date formatted as 'Month Day, Year'.
 *
 * @return std::string The formatted date string.
 */
std::string getDate() {
    // Get the current time
    std::time_t now = std::time(nullptr);
    std::tm* localTime = std::localtime(&now);

    // Create an output stream with appropriate formatting
    std::ostringstream oss;
    oss << std::put_time(localTime, "%B %d, %Y");

    return oss.str();
}

int main() {
    // Example usage of the getDate function
    std::cout << "Current Date: " << getDate() << std::endl;
    return 0;
}
