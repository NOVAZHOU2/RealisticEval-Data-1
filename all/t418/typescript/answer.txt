function lengthOfLIS(nums: number[]): number {
    if (nums.length === 0) return 0;
    const dp: number[] = new Array(nums.length).fill(1);
    for (let i = 0; i < nums.length; i++) {

        }
    }
    return Math.max(...dp);
}
