Here's how you might write the C++ test case:

```cpp
#include <catch2/catch.hpp>
#include <ctime>
#include <iomanip>
#include <sstream>

// Function to get the current date as YYYY-MM-DD
std::string getCurrentDate() {
    std::time_t now = std::time(nullptr);
    std::tm tm = *std::localtime(&now);

    std::ostringstream oss;
    oss << std::put_time(&tm, "%Y-%m-%d");
    return oss.str();
}

TEST_CASE("GetCurrentDate", "[date]") {
    // Get the current date string
    std::string currentDate = getCurrentDate();

    // Check if the date string has the correct format
    REQUIRE(currentDate.size() == 10); // YYYY-MM-DD

    // Extract year, month, and day from the date string
    int year = std::stoi(currentDate.substr(0, 4));
    int month = std::stoi(currentDate.substr(5, 2));
    int day = std::stoi(currentDate.substr(8, 2));

    // Check if the extracted values are valid
    REQUIRE(year > 1970); // Assuming we don't need dates before 1970
    REQUIRE(month >= 1 && month <= 12);
    REQUIRE(day >= 1 && day <= 31);
}
```
