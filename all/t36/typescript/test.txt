Here's how you can convert your Python unittest test class to TypeScript using Jest:

```typescript
import { describe, it, expect } from '@jest/globals';

function floydWarshallShortestPaths(adjacencyMatrix: number[][]): number[][] {
    // ... implementation remains same ...
}

describe('floydWarshallShortestPaths', () => {
    it('should return correct shortest path matrix for given adjacency matrix', () => {
        const inputMatrix = [
            [0, 3, float('inf'), 7],
            [8, 0, 2, float('inf')],
            [5, float('inf'), 0, 1],
            [2, float('inf'), float('inf'), 0]
        ];
        
        const expectedOutput = [
            [0, 3, 6, 7],
            [8, 0, 2, 9],
            [5, 4, 0, 1],
            [2, 5, 6, 0]
        ];

        const output = floydWarshallShortestPaths(inputMatrix);
        expect(output).toEqual(expectedOutput);
    });

    it('should handle graphs with negative weights correctly', () => {
        const inputMatrix = [
            [0, -1, float('inf')],
            [float('inf'), 0, 3],
            [2, float('inf'), 0]
        ];
        
        const expectedOutput = [
            [0, -1, 2],
            [float('inf'), 0, 3],
            [2, float('inf'), 0]
        ];

        const output = floydWarshallShortestPaths(inputMatrix);
        expect(output).toEqual(expectedOutput);
    });
});
```

Please note that in JavaScript/TypeScript, `float('inf')` doesn't exist. Instead, we usually use `Infinity`. Also, make sure to install and set up Jest in your project before running these tests.
