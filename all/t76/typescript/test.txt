import { describe, it, expect } from '@jest/globals';

function removeCommonIndentation(multilineText: string): string {
    /**
     * Removes the common leading indentation from each line in a given multi-line string,
     * preserving the relative indentation of the text.
     *
     * @param {string} multilineText - The input string containing multiple lines.
     * @returns {string} - The sanitized string with common leading indentation removed.
     */
    const lines = multilineText.split('\n');
    if (lines.length === 0) return '';

    // Find the minimum non-empty line's indent level
    let minIndentLevel = Infinity;
    for (const line of lines) {
        const trimmedLine = line.trim();
        if (trimmedLine !== '') {
            const indentLevel = line.length - trimmedLine.length;
            if (indentLevel < minIndentLevel) {
                minIndentLevel = indentLevel;
            }
        }
    }

    // Remove the common indentation from each line
    return lines.map(line => line.slice(minIndentLevel)).join('\n');
}

describe('removeCommonIndentation', () => {
    it('should remove common leading indentation', () => {
        const input = `
            function example() {
                console.log("Hello");
                console.log("World");
            }
        `;
        const expectedOutput = `function example() {
console.log("Hello");
console.log("World");
}`;
        expect(removeCommonIndentation(input)).toBe(expectedOutput);
    });

    it('should handle empty strings', () => {
        const input = '';
        const expectedOutput = '';
        expect(removeCommonIndentation(input)).toBe(expectedOutput);
    });

    it('should handle strings without leading indentation', () => {
        const input = 'function example() {\n    console.log("Hello");\n}';
        const expectedOutput = 'function example() {\n    console.log("Hello");\n}';
        expect(removeCommonIndentation(input)).toBe(expectedOutput);
    });
});
