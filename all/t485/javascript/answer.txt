```javascript
function prepareQuery(sql, params) {
    const regex = /\$[a-zA-Z_][a-zA-Z0-9_]*/g;
    let matches = sql.match(regex);
    
    if (!matches || !params) return [sql, []];

    let newSql = sql;
    let paramValues = [];

    matches.forEach((match, index) => {
        let paramName = match.slice(1); // Remove the '$' character
        if (params.hasOwnProperty(paramName)) {
            newSql = newSql.replace(match, `$${index + 1}`);
            paramValues.push(params[paramName]);
        } else {
            console.warn(`Parameter ${paramName} not found in params`);
        }
    });

    return [newSql, paramValues];
}
```

This JavaScript function works similarly to your Python function. It takes an SQL query string and a parameter object as inputs and returns a tuple containing the modified SQL query string with positional parameters and a list of parameter values sorted according to their new positions in the query. If a parameter from the `params` object is not found in the SQL query, it logs a warning message.
