std::string compressHash(const std::string& hash) {

    const int base = 62;
    const std::string alphabet = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";

    // Initialize the result string
    std::string result;

    while (result.length() < 5) {

    }

}
