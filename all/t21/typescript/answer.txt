Here's how you could translate that Python code into TypeScript:

```typescript
import fs from 'fs';
import { difference } from 'lodash';

interface FileCompareResult {
  differences: string[];
}

export function compareFiles(file1Path: string, file2Path: string): Promise<FileCompareResult> {
  return new Promise((resolve, reject) => {
    fs.readFile(file1Path, 'utf8', (err, data1) => {
      if (err) {
        return reject(err);
      }

      fs.readFile(file2Path, 'utf8', (err, data2) => {
        if (err) {
          return reject(err);
        }

        const differences = difference(data1.split('\n'), data2.split('\n'));
        
        resolve({differences});
      });
    });
  });
}
```

Please note that in TypeScript, we need to use promises instead of exceptions for handling errors, so I've converted your function to return a promise. Also, because TypeScript doesn't have built-in support for unified diff format, I've used lodash's `difference` method to find the differences between the two files' content line-by-line. 

Make sure to install lodash before running this code with `npm install lodash`.
