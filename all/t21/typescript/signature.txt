```typescript
import fs from 'fs';
import { difference } from 'lodash';

/**
 * Compare the contents of two files and return the differences in unified diff format.
 *
 * @param file1Path - Path to the first file.
 * @param file2Path - Path to the second file.
 * @returns An array containing the lines of differences, if any.
 * @throws {Error} If either file does not exist or there is an error reading the files.
 */
function compareFiles(file1Path: string, file2Path: string): string[] {
    // Read the contents of both files
    let file1Content = '';
    let file2Content = '';

    try {
        file1Content = fs.readFileSync(file1Path, 'utf8');
        file2Content = fs.readFileSync(file2Path, 'utf8');
    } catch (error) {
        throw new Error(`Failed to read one or both files: ${error.message}`);
    }

    // Split the content into lines
    const lines1 = file1Content.split('\n');
    const lines2 = file2Content.split('\n');

    // Find the differences using lodash's difference function
    const diffLines = difference(lines1, lines2);

    return diffLines;
}
```

### Explanation:
- **Imports**: The `fs` module is used for file system operations, and `difference` from `lodash` is used to find the differences between arrays.
- **Function Signature**: The TypeScript function signature matches the Python function signature closely, including parameter types and return type.
- **File Reading**: The `fs.readFileSync` method reads the content of the files synchronously. If an error occurs during reading, it throws an error with a descriptive message.
- **Line Splitting**: The file content is split into lines using the `split('\n')` method.
- **Finding Differences**: The `difference` function from `lodash` is used to find the lines that are present in `lines1` but not in `lines2`.
- **Return Value**: The function returns an array of strings representing the lines of differences.
