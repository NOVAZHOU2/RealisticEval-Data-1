Here's how you can convert your Python code to JavaScript using Node.js:

```javascript
const fs = require('fs');
const yaml = require('js-yaml');

function readYAML(filePath) {
  return new Promise((resolve, reject) => {
    fs.readFile(filePath, 'utf8', (err, data) => {
      if (err) {
        if (err.code === 'ENOENT') {
          reject(new Error('File not found'));
        } else {
          reject(err);
        }
      } else {
        try {
          resolve(yaml.load(data));
        } catch (e) {
          reject(e);
        }
      }
    });
  });
}

module.exports = { readYAML };
```

This JavaScript function `readYAML` reads a YAML file asynchronously and returns its content as a JavaScript object or array. It uses Node.js' built-in `fs` module to handle file reading and the `js-yaml` package for parsing the YAML data. This package needs to be installed via npm before running the script (`npm install js-yaml`).
