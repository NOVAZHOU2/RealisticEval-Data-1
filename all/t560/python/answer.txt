def get_line_number(content: str, index: int) -> int:
    """
    Gets the line number in the content at the specified index.

    :param content: The string content to check.
    :param index: The character index to find the line number for.
    :return: The line number corresponding to the given index.
    """
    # Split the content into lines
    lines = content.split('\n')
    
    # Initialize variables to track the current line and its length
    current_line = 1
    current_length = 0
    
    # Iterate through each line
    for line in lines:
        # Check if the index is within the bounds of the current line
        if current_length <= index < current_length + len(line):
            return current_line
        
        # Move to the next line
        current_length += len(line) + 1  # Add 1 for the newline character
        current_line += 1
    
    # If the index is beyond the last line, return the total number of lines
    return current_line
