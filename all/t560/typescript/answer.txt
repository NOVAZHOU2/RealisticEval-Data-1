/**
 * Gets the line number in the content at the specified index.
 *
 * @param {string} content - The string content to check.
 * @param {number} index - The character index to find the line number for.
 * @returns {number} - The line number corresponding to the given index.
 */
function getLineNumber(content: string, index: number): number {
    // Split the content into lines using the newline character
    const lines = content.split('\n');
    
    // Initialize variables to track the current position and line number
    let currentPosition = 0;
    let lineNumber = 1;
    
    // Iterate through each line
    for (let i = 0; i < lines.length; i++) {
        // Add the length of the current line plus one (for the newline character)
        currentPosition += lines[i].length + 1;
        
        // If the index is within the current line, return the current line number
        if (index < currentPosition) {
            return lineNumber;
        }
        
        // Increment the line number
        lineNumber++;
    }
    
    // If the index is beyond the last line, return the total number of lines
    return lineNumber;
}
