import java.util.regex.Pattern;
import java.util.regex.Matcher;

/**
 * Converts HTML heading tags (h1-h6) to corresponding Markdown headings.
 *
 * @param html The HTML string containing headings.
 * @return The converted Markdown string.
 */
public class HtmlToMarkdownConverter {

    public static String convertHtmlHeadingsToMarkdown(String html) {
        // Replace h1 to h6 tags with corresponding Markdown headings
        html = replaceHtmlHeading(html, "<h1>", "# ");
        html = replaceHtmlHeading(html, "<h2>", "## ");
        html = replaceHtmlHeading(html, "<h3>", "### ");
        html = replaceHtmlHeading(html, "<h4>", "#### ");
        html = replaceHtmlHeading(html, "<h5>", "##### ");
        html = replaceHtmlHeading(html, "<h6>", "###### ");

        return html;
    }

    private static String replaceHtmlHeading(String input, String tag, String replacementStart) {
        Pattern pattern = Pattern.compile(tag + "(.*?)" + "</" + tag.substring(1) + ">");
        Matcher matcher = pattern.matcher(input);
        return matcher.replaceAll(replacementStart + "$1");
    }

    public static void main(String[] args) {
        String html = "<h1>Heading 1</h1><h2>Heading 2</h2><h3>Heading 3</h3>";
        String markdown = convertHtmlHeadingsToMarkdown(html);
        System.out.println(markdown);
    }
}
