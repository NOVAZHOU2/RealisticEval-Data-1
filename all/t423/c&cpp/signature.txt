```cpp
#include <fstream>
#include <string>
#include <iostream>

// Writes a line to a text file only if the line with the same content does not already exist.
void write_unique_line_to_file(const std::string& filename, const std::string& line_content) {
    // Check if the line already exists in the file
    std::ifstream infile(filename);
    if (infile) {
        std::string existing_line;
        while (std::getline(infile, existing_line)) {
            if (existing_line == line_content) {
                // Line already exists, do nothing
                return;
            }
        }
        infile.close();
    }

    // Write the line to the file
    std::ofstream outfile;
    outfile.open(filename, std::ios_base::app); // Open in append mode
    if (outfile.is_open()) {
        outfile << line_content << std::endl;
        outfile.close();
    } else {
        std::cerr << "Unable to open file: " << filename << std::endl;
    }
}
```

This function checks if the specified line already exists in the file before writing it. If the line does not exist, it appends the content to the end of the file. If the file cannot be opened, it outputs an error message.
