Below is the equivalent TypeScript code using Jest for testing:

```typescript
import fs from 'fs';
import { resolve } from 'path';

let filename: string = resolve('temp.txt');

function writeUniqueLineToFile(filename: string, lineContent: string): void {
    /**
     * Writes a line to a text file only if the line with the same content does not already exist.
     *
     * @param {string} filename - The name of the file to write to.
     * @param {string} lineContent - The content of the line to write.
     */
    fs.readFile(filename, 'utf8', (err, data) => {
        if (err && err.code === 'ENOENT') {
            // File doesn't exist, write the line directly
            fs.writeFile(filename, lineContent + '\n', (err) => {
                if (err) throw err;
            });
        } else {
            const lines = data.split('\n');
            if (!lines.includes(lineContent)) {
                fs.appendFile(filename, lineContent + '\n', (err) => {
                    if (err) throw err;
                });
            }
        }
    });
}

describe('writeUniqueLineToFile', () => {
    beforeAll(() => {
        // Create a temporary file for testing
        fs.writeFileSync(filename, '');
    });

    afterAll(() => {
        // Clean up the temporary file
        fs.unlinkSync(filename);
    });

    it('should write a new unique line to the file', () => {
        writeUniqueLineToFile(filename, 'Hello, world!');
        const fileContent = fs.readFileSync(filename, 'utf8');
        expect(fileContent.trim()).toBe('Hello, world!');
    });

    it('should not write a duplicate line to the file', () => {
        writeUniqueLineToFile(filename, 'Hello, world!');
        writeUniqueLineToFile(filename, 'Another line');
        const fileContent = fs.readFileSync(filename, 'utf8');
        expect(fileContent.trim()).toBe('Hello, world!\nAnother line');
    });

    it('should append multiple unique lines to the file', () => {
        writeUniqueLineToFile(filename, 'Yet another line');
        const fileContent = fs.readFileSync(filename, 'utf8');
        expect(fileContent.trim()).toBe('Hello, world!\nAnother line\nYet another line');
    });
});
```

In this code snippet, we have created a function `writeUniqueLineToFile` which checks if the line already exists in the file before writing it. We also have Jest test cases that set up and tear down a temporary file for testing purposes. Each test case then verifies the expected behavior of the function by checking the contents of the file after calling `writeUniqueLineToFile`.
