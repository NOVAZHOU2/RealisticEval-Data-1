const fs = require('fs');

function writeUniqueLineToFile(filename, lineContent) {
    /**
     * Writes a line to a text file only if the line with the same content does not already exist.
     *
     * @param {string} filename - The name of the file to write to.
     * @param {string} lineContent - The content of the line to write.
     */

    // Check if the lineContent already exists in the file
    fs.readFile(filename, 'utf8', (err, data) => {
        if (err && err.code === 'ENOENT') {
            // File does not exist, so create it and write the line
            fs.writeFile(filename, lineContent + '\n', (writeErr) => {
                if (writeErr) {
                    console.error(`Error writing to '${filename}':`, writeErr);
                } else {
                    console.log(`Line '${lineContent}' successfully written to '${filename}'.`);
                }
            });
        } else if (err) {
            console.error(`Error reading from '${filename}':`, err);
        } else {
            // File exists, check if the lineContent is already in the file
            if (data.includes(lineContent)) {
                console.log(`Line '${lineContent}' already exists in '${filename}'. Not writing again.`);
            } else {
                // Append the lineContent to the file
                fs.appendFile(filename, lineContent + '\n', (appendErr) => {
                    if (appendErr) {
                        console.error(`Error appending to '${filename}':`, appendErr);
                    } else {
                        console.log(`Line '${lineContent}' successfully written to '${filename}'.`);
                    }
                });
            }
        }
    });
}
