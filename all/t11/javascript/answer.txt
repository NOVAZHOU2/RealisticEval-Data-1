class TrieNode {
    constructor() {
        this.children = {};
    }
}

class Trie {
    /**
     * Implement a dictionary tree for fast string retrieval and storage
     */

    insert(word) {
        let node = this.root || (this.root = new TrieNode());
        for (const char of word) {
            if (!node.children[char]) {
                node.children[char] = new TrieNode();
            }
            node = node.children[char];
        }
    }

    search(word) {
        let node = this.root;
        for (const char of word) {
            if (!node || !node.children[char]) {
                return false;
            }
            node = node.children[char];
        }
        return true;
    }

    startsWith(prefix) {
        let node = this.root;
        for (const char of prefix) {
            if (!node || !node.children[char]) {
                return false;
            }
            node = node.children[char];
        }
        return true;
    }
}
