```typescript
/**
 * Checks whether the provided password meets the specified format requirements:
 * - At least one number
 * - At least one lowercase letter
 * - At least one uppercase letter
 * - At least one punctuation mark
 * - Minimum length of 8 characters
 *
 * @param {string} password - The password string to validate
 * @returns {boolean} Returns true if the password meets all requirements; otherwise, false.
 */
function isValidPassword(password: string): boolean {
    // Regular expression to check for at least one number, one lowercase letter,
    // one uppercase letter, and one punctuation mark.
    const regex = /^(?=.*\d)(?=.*[a-z])(?=.*[A-Z])(?=.*[\W_]).{8,}$/;
    
    return regex.test(password);
}
```

In this TypeScript version:
- The `password` parameter is explicitly typed as `string`.
- The return value is explicitly typed as `boolean`.
- A regular expression (`regex`) is used to ensure that the password meets all the specified criteria. The `test` method of the regex object is used to check if the password matches the pattern.
