```python
import re

/**
 * Checks whether the provided password meets the specified format requirements:
 * - At least one number
 * - At least one lowercase letter
 * - At least one uppercase letter
 * - At least one punctuation mark
 * - Minimum length of 8 characters
 *
 * @param {str} password - The password string to validate
 * @return {bool} True if the password meets all requirements; otherwise, False.
 */
def is_valid_password(password):
    # Check for minimum length
    if len(password) < 8:
        return False
    
    # Check for at least one number
    if not re.search(r'\d', password):
        return False
    
    # Check for at least one lowercase letter
    if not re.search(r'[a-z]', password):
        return False
    
    # Check for at least one uppercase letter
    if not re.search(r'[A-Z]', password):
        return False
    
    # Check for at least one punctuation mark
    if not re.search(r'[^\w\s]', password):
        return False
    
    return True
```

### Explanation:
- **Minimum Length**: The `len(password) < 8` check ensures that the password has at least 8 characters.
- **At Least One Number**: The `re.search(r'\d', password)` checks if there is at least one digit in the password.
- **At Least One Lowercase Letter**: The `re.search(r'[a-z]', password)` checks if there is at least one lowercase letter.
- **At Least One Uppercase Letter**: The `re.search(r'[A-Z]', password)` checks if there is at least one uppercase letter.
- **At Least One Punctuation Mark**: The `re.search(r'[^\w\s]', password)` checks if there is at least one punctuation mark.
