#include <vector>

// compute the squared Euclidean distance between two vectors.
// 
// Args:
//     vec1: std::vector<int> - First vector.
//     vec2: std::vector<int> - Second vector.
// 
// Returns:
//     int - Euclidean distance between vec1 and vec2.
int squaredEuclideanDistance(const std::vector<int>& vec1, const std::vector<int>& vec2) {
    int distance = 0;
    for (size_t i = 0; i < vec1.size(); ++i) {
        int diff = vec1[i] - vec2[i];
        distance += diff * diff;
    }
    return distance;
}
