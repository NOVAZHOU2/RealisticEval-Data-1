/**
 * Removes the first occurrence of a specified element from an array.
 *
 * @param array - The array from which to remove the element.
 * @param element - The element to remove from the array.
 * @return A new array with the element removed, or the original array if the element is not found.
 */
std::vector<int> removeElementInArray(std::vector<int>& array, int element) {
    auto it = std::find(array.begin(), array.end(), element);
    if (it != array.end()) {
        array.erase(it);
    }
    return array;
}
