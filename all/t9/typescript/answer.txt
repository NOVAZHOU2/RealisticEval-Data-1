function isPointOnLine(A: [number, number], B: [number, number], C: [number, number]): boolean {
    /**
     * Check whether a point is on a line formed by two points,
     * such as point C is on a line formed by points A and B
     * 
     * @param A - point A (x, y)
     * @param B - point B (x, y)

     * @returns is C in line of A B
