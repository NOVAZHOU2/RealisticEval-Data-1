function isPointOnLine(A, B, C) {
  /*
   * Check whether A point is on a line formed by two points, such as point C is on a line formed by points A and B
   * @param {Array} A - point A xy
   * @param {Array} B - point B xy
   * @param {Array} C - point C xy
   * @return {boolean} is C in line of A B
   */
  
  // Calculate the area of triangle ABC
  let area = Math.abs((A[0]*(B[1]-C[1]) + B[0]*(C[1]-A[1]) + C[0]*(A[1]-B[1])) / 2);
  
  // If area is zero, then points A, B, and C are collinear
  if(area == 0){
    return true;
  }
  else{
    return false;
  }
}

This JavaScript function `isPointOnLine` takes three arguments `A`, `B`, and `C`, which represent the coordinates of the points. If the area is zero, it means that the points are collinear, i.e., they lie on the same straight line, and hence the function returns `true`; otherwise, it returns `false`.
