public class EdgeMidpointCalculator {

    /**
     * Calculate the midpoints from a given array of edges.
     * For example:
     *      input: [0, 1, 2]
     *      output: [0.5, 1.5]
     *
     * @param edges An array of edge values.
     * @return An array of midpoints calculated from the edges.
     */
    public static double[] getMidsFromEdges(double[] edges) {
        int length = edges.length - 1;
        double[] mids = new double[length];

        for (int i = 0; i < length; i++) {
            mids[i] = (edges[i] + edges[i + 1]) / 2.0;
        }

        return mids;
    }

    public static void main(String[] args) {
        // Example usage
        double[] edges = {0, 1, 2};
        double[] mids = getMidsFromEdges(edges);

        System.out.print("Mids: ");
        for (double mid : mids) {
            System.out.print(mid + " ");
        }
    }
}
