package org.real.temp;

import java.util.Arrays;

public class Answer {
    
    public static double[] getMidsFromEdges(double[] edges) {
        int n = edges.length;
        double[] mids = new double[n - 1];
        
        for(int i = 0; i < n - 1; i++) {
            mids[i] = (edges[i] + edges[i+1]) / 2.0;
        }
        
        return mids;
    }

    public static void main(String[] args) {
        double[] edges = {0, 1, 2};
        double[] mids = getMidsFromEdges(edges);
        System.out.println(Arrays.toString(mids));
    }
}

- The `getMidsFromEdges` method calculates the midpoints from an array of edge values.
