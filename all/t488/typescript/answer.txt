```typescript
import { execSync } from 'child_process';
import { match } from 'assert';

function getLocalIP(interfaceName: string = 'Wi-Fi'): string | null {
  try {
    const output = execSync(`ipconfig /all`).toString();
    const regex = new RegExp(`${interfaceName}\\s+.*?IPv4 Address\\s+: (.*?)\\r`, 'g');
    let match;
    while ((match = regex.exec(output)) !== null) {
      return match[1];
    }
  } catch (error) {
    console.error(error);
  }
  return null;
}
```

Note: This code uses `execSync` from Node.js' built-in `child_process` module to execute the `ipconfig /all` command in Windows. It then uses a regular expression to extract the IPv4 address associated with the specified interface. If no match is found or an error occurs during execution, it returns `null`. 

Please make sure that you have Node.js installed and running on your machine before executing this code.
