import java.util.Set;
import java.util.HashSet;
import java.util.Arrays;

public class Answer {
    public static boolean areSetsEqual(Set<Double> set1, Set<Double> set2, double rtol, double atol) {
        if (set1.size() != set2.size()) return false;

        Double[] arr1 = set1.toArray(new Double[0]);
        Double[] arr2 = set2.toArray(new Double[0]);

        Arrays.sort(arr1);
        Arrays.sort(arr2);

        for(int i = 0; i < arr1.length; i++) {
            if(Math.abs(arr1[i] - arr2[i]) > Math.max(rtol * Math.abs(arr2[i]), atol)) {
                return false;
            }
        }

        return true;
    }
}
