```typescript
function eightQueens(): void {
    const N: number = 8;
    let board: string[][] = Array.from({ length: N }, () => Array(N).fill('.'));

    function isSafe(row: number, col: number): boolean {
        for (let i = 0; i < col; i++) {
            if (board[row][i] === 'Q') return false;
        }

        for (let i = row, j = col; i >= 0 && j >= 0; i--, j--) {
            if (board[i][j] === 'Q') return false;
        }

        for (let i = row, j = col; i < N && j >= 0; i++, j--) {
            if (board[i][j] === 'Q') return false;
        }

        return true;
    }

    function solveNQueens(col: number): boolean {
        if (col >= N) {
            console.log(printBoard(board));
            return true;
        }

        for (let i = 0; i < N; i++) {
            if (isSafe(i, col)) {
                board[i][col] = 'Q';
                if (solveNQueens(col + 1)) {
                    return true;
                }
                board[i][col] = '.';
            }
        }

        return false;
    }

    function printBoard(board: string[][]): string {
        return board.map(row => row.join(' ')).join('\n');
    }

    if (!solveNQueens(0)) {
        console.log("No solution");
    }
}

eightQueens();
```

### Explanation:
- **`board`:** A 2D array representing the chessboard.
- **`isSafe(row, col)`:** Checks if placing a queen at `board[row][col]` is safe from attacks.
- **`solveNQueens(col)`:** Recursively tries to place queens in each column and backtracks if a placement leads to no valid solutions.
- **`printBoard(board)`:** Converts the board into a string format for printing.
