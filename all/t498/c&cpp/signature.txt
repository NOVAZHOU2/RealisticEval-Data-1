```cpp
#include <openssl/md5.h> // Include OpenSSL library for MD5 hashing
#include <iomanip>       // For std::hex
#include <sstream>     // For std::stringstream
#include <string>      // For std::string

/**
 * Computes and returns the MD5 hash of the input string.
 *
 * @param inputString The string to be hashed
 * @return The MD5 hash of the input string in hexadecimal format
 */
std::string computeMD5(const std::string& inputString) {
    unsigned char digest[MD5_DIGEST_LENGTH]; // Buffer for MD5 hash result

    MD5_CTX context; // Create an MD5 context
    MD5_Init(&context); // Initialize the MD5 context
    MD5_Update(&context, inputString.c_str(), inputString.size()); // Update the context with the input string
    MD5_Final(digest, &context); // Finalize the MD5 computation and store the result in 'digest'

    std::stringstream ss; // Create a stringstream object for formatting
    for(int i = 0; i < MD5_DIGEST_LENGTH; ++i)
        ss << std::hex << std::setw(2) << std::setfill('0') << (int)digest[i]; // Format each byte of the digest into a two-digit hex number

    return ss.str(); // Return the formatted MD5 hash as a string
}
```
This code uses the OpenSSL library for MD5 hashing. If you don't have it installed on your system, you can install it using a package manager or build it from source. Note that you need to link against the OpenSSL library when compiling your program.
