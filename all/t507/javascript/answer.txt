function isStrongPassword(password) {
    // Regular expression for validating a strong password
    const regex = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)[a-zA-Z\d]{8,}$/;

    // Test the input against the regular expression
    return regex.test(password);
}
