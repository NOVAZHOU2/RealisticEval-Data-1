Here is the equivalent JavaScript code using the 'lodash' library:

```javascript
const _ = require('lodash');

function findMinWindowSubstring(source, target) {
    const targetCount = _.countBy(target);
    let requiredChars = Object.keys(targetCount).length;
    let left = 0, right = 0;
    let formed = 0;
    let minLen = Infinity, minWindow = '';
    const windowCounts = {};

    while (right < source.length) {
        const charRight = source[right];
        windowCounts[charRight] = (windowCounts[charRight] || 0) + 1;

        if (targetCount[charRight] && windowCounts[charRight] === targetCount[charRight]) {
            formed++;
        }

        while (left <= right && formed === requiredChars) {
            const charLeft = source[left];

            if (right - left + 1 < minLen) {
                minLen = right - left + 1;
                minWindow = source.substring(left, right + 1);
            }

            windowCounts[charLeft]--;
            if (targetCount[charLeft] && windowCounts[charLeft] < targetCount[charLeft]) {
                formed--;
            }
            left++;
        }
        right++;
    }

    return minWindow;
}
```
