```java
import java.util.Arrays;

/**
 * Converts a list of hole positions to the ring format (array of 1s and 0s).
 *
 * @param holes An array of integers representing the positions of the holes (0-indexed).
 * @return An array of length 32, where positions with holes are 0 and others are 1.
 */
public class HoleConverter {

    public static int[] convertToRingFormat(int[] holes) {
        // Initialize an array of size 32 with all elements set to 1
        int[] ringFormat = new int[32];
        Arrays.fill(ringFormat, 1);

        // Set positions with holes to 0
        for (int hole : holes) {
            if (hole >= 0 && hole < 32) { // Ensure the position is within the valid range
                ringFormat[hole] = 0;
            }
        }

        return ringFormat;
    }
}
```
