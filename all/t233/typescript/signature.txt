```typescript
/**
 * Removes comments from the provided string. Comments start with a '#' and end at the newline.
 * For example:
 *      input: Hello, world! # This is a comment
 *      output:  Hello, world!
 * @param {string} inputString - The input string containing potential comments.
 * @returns {string} - The string with all comments removed.
 */
function removeComments(inputString: string): string {
    // Split the input string into lines
    const lines = inputString.split('\n');
    
    // Process each line to remove comments
    const cleanedLines = lines.map(line => {
        // Find the index of the first '#' character
        const hashIndex = line.indexOf('#');
        
        // If there's a '#', return the part before it; otherwise, return the original line
        return hashIndex !== -1 ? line.substring(0, hashIndex).trim() : line;
    });
    
    // Join the cleaned lines back into a single string
    return cleanedLines.join('\n');
}
```

This TypeScript function `removeComments` does the same thing as your Python function: it removes any comments from a given string where comments start with `#` and continue until the end of the line.
