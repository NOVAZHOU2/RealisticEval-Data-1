export function getPalindromeList(n: number): number[] {
    const isPalindrome = (num: number): boolean => {
        const strNum = num.toString();
        return strNum === strNum.split('').reverse().join('');
    };

    const result: number[] = [];
    for (let i = 0; i <= n; i++) {
        if (isPalindrome(i)) {
            result.push(i);
        }
    }
    return result;
}
