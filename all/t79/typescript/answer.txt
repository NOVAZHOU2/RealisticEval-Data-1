```typescript
function dateRangeString(startDate: string, endDate: string): string {
    const startDateObj = new Date(startDate);
    const endDateObj = new Date(endDate);

    const months = ['January', 'February', 'March', 'April', 'May', 'June', 
                   'July', 'August', 'September', 'October', 'November', 'December'];

    const startMonthName = months[startDateObj.getMonth()];
    const endMonthName = months[endDateObj.getMonth()];

    const startDay = startDateObj.getDate();
    const endDay = endDateObj.getDate();

    const startYear = startDateObj.getFullYear();
    const endYear = endDateObj.getFullYear();

    let result = '';

    if (startMonthName === endMonthName && startYear === endYear) {
        result = `${startMonthName} ${startDay} to ${endDay}, ${startYear}`;
    } else if (startYear === endYear) {
        result = `${startMonthName} ${startDay} to ${endMonthName} ${endDay}, ${startYear}`;
    } else {
        result = `${startMonthName} ${startDay}, ${startYear} to ${endMonthName} ${endDay}, ${endYear}`;
    }

    return result;
}
```

This function does the same thing as your Python function but it's written in TypeScript. It takes two strings representing dates, converts them into `Date` objects, and then generates a string based on whether they are in the same month or year.
