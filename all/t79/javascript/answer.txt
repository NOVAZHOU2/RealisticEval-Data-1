```javascript
function dateRangeString(startDate, endDate) {
    /**
     * Generates a string based on the entered start and end dates. If the start date and end date are in the same month,
     * only one month will be displayed. If not, the start and end months will be displayed separately.
     * 
     * @param {string} startDate - The start date in 'YYYY-MM-DD' format.
     * @param {string} endDate - The end date in 'YYYY-MM-DD' format.
     * @returns {string} - A string representing the date range in a human-readable format.
     */

    // Convert the date strings to Date objects
    const start = new Date(startDate);
    const end = new Date(endDate);

    // Get the year, month, day from the start date
    const startYear = start.getFullYear();
    const startMonth = start.getMonth() + 1; // getMonth() returns 0-11
    const startDay = start.getDate();

    // Get the year, month, day from the end date
    const endYear = end.getFullYear();
    const endMonth = end.getMonth() + 1;
    const endDay = end.getDate();

    // Format the month names
    const monthNames = ["January", "February", "March", "April", "May", "June",
                        "July", "August", "September", "October", "November", "December"];
    const formattedStartMonth = monthNames[startMonth - 1];
    const formattedEndMonth = monthNames[endMonth - 1];

    // Check if the start and end dates are in the same month and year
    if (startYear === endYear && startMonth === endMonth) {
        return `${formattedStartMonth} ${startDay} to ${endDay}, ${startYear}`;
    } else {
        return `${formattedStartMonth} ${startDay}, ${startYear} to ${formattedEndMonth} ${endDay}, ${endYear}`;
    }
}

// Example usage:
console.log(dateRangeString("2023-08-01", "2023-08-15")); // Output: "August 1 to 15, 2023"
```

This JavaScript function `dateRangeString` takes two date strings as input, converts them into `Date` objects, and then generates a human-readable date range string based on whether the start and end dates fall within the same month or different months.
