package org.real.temp;

public class Answer {

    public static Integer[] findPowers(int num) {
        if (num == 0)
            return null;
        
        int n = 0;
        while ((num & 1) == 0) {
            num >>= 1;
            n++;
        }

        int m = 0;
        while (num % 3 == 0) {
            num /= 3;
            m++;
        }
        
        if (num != 1)
            return null;
        
        return new Integer[]{n, m};
    }
}
